#include <stdio.h>


int strlen(char str[]){
    int i=0, count;
    char c = str[i];
    while(c!='\0'){
        c = str[i];
        i++;
    } 
    count = i-1;
    return count;

}
int main(){
    char str[] = "Harry bhai";
    
    printf("%d", strlen(str));
    return 0;
}

########### Chapter 8 - Practice Set/05_problem5.c ###########


#include <stdio.h>

int mystrlen(char str[])
{
    int i = 0, count;
    char c = str[i];
    while (c != '\0')
    {
        c = str[i];
        i++;
    }
    count = i - 1;
    return count;
}

void mystrcpy(char target[], char source[])
{
    for (int i = 0; i < mystrlen(source); i++)
    {
        target[i] = source[i];
    }
    target[mystrlen(source)] = '\0';
}


int main()
{
    char source[] = "harry";
    char target[30];
    mystrcpy(target, source); // target now contains "harry"
    printf("%s %s", source, target);
    return 0;
}
